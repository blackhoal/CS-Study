# 2. 구현
```python
def solution(phone_book):
    flag = True
    
    phone_book.sort()
    
    for i in range(len(phone_book) - 1):
        if len(phone_book[i]) < len(phone_book[i + 1]) :
            next_length = len(phone_book[i])
            if phone_book[i] == phone_book[i + 1][:next_length] :
                flag = False
    
    return flag
```
