## ① Python

def solution(lottos, win_nums):
    zero_cnt = lottos.count(0)
    cnt = 0

    lottos.sort()
    win_nums.sort()

    for i in lottos[zero_cnt:] :
        if i in win_nums :
            cnt += 1

    if cnt < 2 :
        min_rank = 6
    else :
        min_rank = 7 - cnt

    max_rank = min_rank - zero_cnt
    if max_rank < 2 :
        max_rank = 1
    answer = [max_rank, min_rank]
    return answer
    
// 타인 작성 답안
def solution(lottos, win_nums):
    cnt = len(set(lottos) & set(win_nums))
    zero = lottos.count(0)
    
    return [7-max(cnt+zero,1) ,7-max(cnt,1)]
